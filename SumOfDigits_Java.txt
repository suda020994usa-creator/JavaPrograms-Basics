import java.util.*;

class SumofDigits {
    public static void main(String[] args) {
     int sum =0;
     int n = 123456;
     while(n!=0)
     {
         sum +=n%10;
         n/=10;
         
     }
     System.out.println("Sum of digits");
     System.out.println(sum);
      
        
        
    }
}
