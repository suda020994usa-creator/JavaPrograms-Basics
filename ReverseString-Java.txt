class ReverseString
{
    public static void main(String args[])
    {
        String str = "natwest";
        String rev = "";
        for(int i = str.length()-1;i>=0;i--)
        {
            rev = rev + str.charAt(i);
        }
        System.out.println("The Reversed String is ");
        System.out.println(rev);
    }
}
