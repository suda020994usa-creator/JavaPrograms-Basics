class ReverseString
{
    public static void main(String args[])
    {
        String str = "AMMA";
        String rev = "";
        for(int i = str.length()-1;i>=0;i--)
        {
            rev = rev + str.charAt(i);
        }
        System.out.println(rev);
        if(rev.equals(str))
        {
            System.out.println("The String is palindrome");
            
        }
        else
        {
            System.out.println("The String is not palindrome"); 
        }
      
    }
}
