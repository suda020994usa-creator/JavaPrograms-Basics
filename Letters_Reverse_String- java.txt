import java.util.*;
class StringPermutations {
    public static void main(String[] args) {
       String str = "ABC IS DEF";
      
       String[] ptr = str.split(" ");
       ptr[0] = new StringBuilder(ptr[0]).reverse().toString();
       ptr[1] = new StringBuilder(ptr[1]).reverse().toString();
       ptr[2] = new StringBuilder(ptr[2]).reverse().toString();
       String res = String.join(" ", ptr);
       System.out.println(res);

       
       
        
    }
}
